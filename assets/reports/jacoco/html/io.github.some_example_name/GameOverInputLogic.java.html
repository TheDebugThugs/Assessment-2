package io.github.some_example_name;

public class GameOverInputLogic {

    public enum Action {
        NONE,
        GO_TO_MENU,
        GO_TO_LEADERBOARD
    }

    public Action handleInput(boolean spacePressed, boolean lPressed) {
        if (spacePressed) {
            return Action.GO_TO_MENU;
        }
        if (lPressed) {
            return Action.GO_TO_LEADERBOARD;
        }
        return Action.NONE;
    }
}
