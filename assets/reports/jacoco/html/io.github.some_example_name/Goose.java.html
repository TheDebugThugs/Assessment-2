package io.github.some_example_name;

import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.math.Rectangle;
import com.badlogic.gdx.math.Vector2;

/**
 * <code> Goose </code> is an enemy that follows the player until fed,
 * stopping when close enough.
 */
public class Goose {

    private Vector2 position;
    private Vector2 startPosition;
    private Vector2 velocity;
    private Texture texture;
    private Player player;
    private GameScreen gameScreen;
    private float speed = 40f;
    private boolean repelled;
    private static float stop_distance = 32f;
    private boolean fed;

    /**
     * Constructor for <code> Goose </code>, with a set of coordinates.
     * @param x Horizontal position for goose to spawn in.
     * @param y Vertical position for goose to spawn in.
     * @param player Player class to follow.
     * @param gameScreen Screen rendering the game.
     */
    public Goose(float x, float y, Player player, GameScreen gameScreen) {
        this(x, y, player, gameScreen, true);
    }

    /**
     * Test-safe constructor (skips texture loading).
     */
    public Goose(float x, float y, Player player, GameScreen gameScreen, boolean loadTexture) {
        this.position = new Vector2(x, y);
        this.startPosition = new Vector2(x, y);
        this.player = player;
        this.gameScreen = gameScreen;
        this.velocity = new Vector2();
        this.repelled = false;
        this.fed = false;

        if (loadTexture) {
            this.texture = new Texture("goose.png");
        }
    }

    /**
     * Update position of goose to get closer to player's new position.
     * @param delta Time elapsed since last update.
     */
    public void set_distance() {
        stop_distance = 5f;
    }

    public void update(float delta) {
        Vector2 direction = new Vector2(player.getPosition()).sub(position);
        float distance = position.dst(player.getPosition());

        direction.nor();

        if (distance > stop_distance && !fed) {
            float newX = position.x + direction.x * speed * delta;
            float newY = position.y + direction.y * speed * delta;

            if (!gameScreen.isCellBlocked(newX, newY)) {
                position.set(newX, newY);
            }

            tryMoveDiagonally(delta, direction);
        }
    }

    public boolean caughtPlayer(Player player) {
        return getPosition().dst(player.getPosition()) <= stop_distance;
    }

    public void seedsUsed() {
        fed = true;
    }

    /**
     * Attempt to move goose diagonally towards player if moving
     * in a straight line is not possible.
     */
    private void tryMoveDiagonally(float delta, Vector2 direction) {
        float newX = position.x + direction.x * speed * delta;
        if (!gameScreen.isCellBlocked(newX, position.y)) {
            position.x = newX;
            return;
        }

        float newY = position.y + direction.y * speed * delta;
        if (!gameScreen.isCellBlocked(position.x, newY)) {
            position.y = newY;
        }
    }

    /**
     * Draw goose sprite.
     */
    public void render(SpriteBatch batch) {
        if (!repelled && texture != null) {
            batch.draw(texture, position.x, position.y, 16, 16);
        }
    }

    /**
     * Return 2D coordinates of goose.
     */
    public Vector2 getPosition() {
        return position;
    }

    /**
     * Return rectangle representing collision bounds of goose.
     */
    public Rectangle getBounds() {
        return new Rectangle(position.x, position.y, 16, 16);
    }

    /**
     * Dispose goose texture.
     */
    public void dispose() {
        if (texture != null) {
            texture.dispose();
        }
    }
}
