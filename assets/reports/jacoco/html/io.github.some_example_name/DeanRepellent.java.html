package io.github.some_example_name;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Input;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.math.Vector2;

public class DeanRepellent {

    //private boolean collected = false;
    //private boolean usable = false;
    //private boolean eventTriggered = false; // ensures hiddenEvents increments once

    private final Texture bottleTexture;
    private final Vector2 position;
    private final DeanRepellentLogic logic = new DeanRepellentLogic();

    public DeanRepellent(float x, float y) {
        bottleTexture = new Texture("Dean-Repellent.png");
        position = new Vector2(x, y);
    }

    /**
     * Update the bottle state.
     *
     * @param player Player reference for distance checking
     * @param dean Dean reference for repelling
     * @param caught Number of times Dean caught the player
     * @param delta Delta time (unused here but may be used later)
     * @param gameScreen GameScreen reference to notify about hidden events
     */
    public void update(Player player, Dean dean, int caught, float delta, GameScreen gameScreen) {
        float distance = player.getPosition().dst(position);
        logic.discover(distance);
        boolean ePressed = Gdx.input.isKeyJustPressed(Input.Keys.E);
        boolean wasCollected = logic.isCollected();

        logic.pickUp(distance, ePressed);

        if (!wasCollected && logic.isCollected()) {
            dean.setRepelled(true);
            gameScreen.onHiddenEventCollected();
        }
    }

    public void render(SpriteBatch batch) {
        if (!logic.isCollected() && logic.isUsable()) {
            batch.draw(bottleTexture, position.x, position.y, 15, 15);
        }
    }

    public boolean isCollected() {
        return logic.isCollected();
    }

    public void dispose() {
        bottleTexture.dispose();
    }
}
