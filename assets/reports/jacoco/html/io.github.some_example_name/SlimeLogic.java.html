package io.github.some_example_name;

import com.badlogic.gdx.math.Rectangle;

public class SlimeLogic {
    private final Rectangle slimeBounds;
    private boolean slimeCheck = false;

    public SlimeLogic(Rectangle slimeBounds) {
        this.slimeBounds = slimeBounds;
    }

    public void update(Rectangle playerBounds) {
        slimeCheck = playerBounds.overlaps(slimeBounds);
    }

    public boolean isSlimeCheck() {
        return slimeCheck;
    }
}
