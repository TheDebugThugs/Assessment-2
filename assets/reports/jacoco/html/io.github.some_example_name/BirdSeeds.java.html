package io.github.some_example_name;

import com.badlogic.gdx.graphics.OrthographicCamera;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.math.Vector2;

/**
 * <code> BirdSeeds </code> represents bird seeds that can be bought and used
 * to interact with the goose. Contains both world sprite and UI icon logic.
 */
public class BirdSeeds {

    private Vector2 position;
    private Texture texture;
    private boolean isUsed;
    private boolean isBought;

    /**
     * Constructor for <code> BirdSeeds </code>, spawning the bird seeds sprite
     * in the given world coordinates. Uses "birdseeds.png" in assets folder.
     *
     * @param x Horizontal position in world
     * @param y Vertical position in world
     */
    public BirdSeeds(float x, float y) {
        this.position = new Vector2(x, y);
        this.texture = new Texture("birdseeds.png");
        this.isUsed = false;
        this.isBought = false;
    }

    /**
     * Test-safe constructor that avoids loading textures.
     * Used only for unit testing.
     *
     * @param x Horizontal position
     * @param y Vertical position
     * @param testMode disables LibGDX texture loading
     */
    public BirdSeeds(float x, float y, boolean testMode) {
        this.position = new Vector2(x, y);
        this.texture = null; // no LibGDX dependency in tests
        this.isUsed = false;
        this.isBought = false;
    }

    /**
     * Render bird seeds sprite within given sprite batch.
     *
     * @param batch SpriteBatch to render world sprite in
     */
    public void render(SpriteBatch batch) {
        if (isUsed && texture != null) {
            batch.draw(texture, position.x, position.y, 16, 16);
        }
    }

    /**
     * Render bird seeds icon in UI when bought.
     *
     * @param batch SpriteBatch to render UI sprite in
     * @param camera Camera connected to batch
     */
    public void renderAsIcon(SpriteBatch batch, OrthographicCamera camera) {
        if (isBought && texture != null) {
            float iconSize = 32f;
            float padding = 20f;

            float cornerX = camera.position.x + (camera.viewportWidth / 2 * camera.zoom);
            float cornerY = camera.position.y + (camera.viewportHeight / 2 * camera.zoom);

            float iconX = cornerX - iconSize - padding;
            float iconY = cornerY - iconSize - padding;

            batch.draw(texture, iconX - 30, iconY, iconSize, iconSize);
        }
    }

    /**
     * Return bird seeds position.
     *
     * @return 2D Vector position
     */
    public Vector2 getPosition() {
        return position;
    }

    /**
     * Returns whether bird seeds have been bought.
     *
     * @return true if bought
     */
    public boolean isBought() {
        return isBought;
    }

    /**
     * Mark bird seeds as bought.
     */
    public void bought() {
        this.isBought = true;
    }

    /**
     * Mark bird seeds as used.
     */
    public void used() {
        this.isUsed = true;
    }

    /**
     * Dispose texture safely.
     */
    public void dispose() {
        if (texture != null) {
            texture.dispose();
        }
    }
}
