package io.github.some_example_name;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Input;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;

public class Textbox {
    private Texture texture;
    private float y;
    private float speed = 2000f;
    private boolean visible = false;
    private float targetY = 20f;
    private static Textbox activeTextbox;

    public Textbox(String image) {
        texture = new Texture(image);
        y = -texture.getHeight();
    }

    public void show() {
        if (activeTextbox != null && activeTextbox != this) {
            activeTextbox.invisible();
        }
        activeTextbox = this;
        visible = true;
    }
    public void invisible() {
        visible = false;
    }

    public void update(float delta) {
        if (visible) {
            if (y < targetY) {
                y += speed * delta;
                if (y > targetY) y = targetY;
            }
        }

            if (Gdx.input.isKeyJustPressed(Input.Keys.SPACE)) {
                visible = false;
            }
        }

    public void draw(SpriteBatch batch, float screenWidth) {
        float scale_x = 0.4f;
        float scale_y = 0.7f;
        float x = (screenWidth - texture.getWidth() * scale_x) / 2f;
        if (visible) {
            batch.draw(texture, x, y, texture.getWidth() * scale_x, texture.getHeight() * scale_y);
        }
    }
    public boolean isVisible() {
        return visible;
    }

    public void dispose() {
        texture.dispose();
    }
}
