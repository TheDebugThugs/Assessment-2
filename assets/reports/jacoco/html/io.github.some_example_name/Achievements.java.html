package io.github.some_example_name;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Input;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;

public class Achievements {
    private Texture texture;
    private float y;
    private float speed = 2000f;
    private boolean visible = false;
    private float targetY = 20f;
    private boolean unlocked = false; // tracks if achievement is unlocked
    private static Achievements activeAchievement;

    public Achievements(String image) {
        texture = new Texture(image);
        y = -texture.getHeight();
    }

    public void unlock() { // unlock achievement and show it
        if (!unlocked) {
            unlocked = true;
            show();
        }
    }

    public void show() {
        if (activeAchievement != null && activeAchievement != this) {
            activeAchievement.invisible();
        }
        activeAchievement = this;
        visible = true;
    }

    public void invisible() {
        visible = false;
    }

    public void update(float delta) {
        if (visible) {
            if (y < targetY) {
                y += speed * delta;
                if (y > targetY) y = targetY;
            }

            if (Gdx.input.isKeyJustPressed(Input.Keys.SPACE)) {
                visible = false;
            }
        }
    }

    public void draw(SpriteBatch batch, float screenWidth) {
        float scaleX = 0.2f;
        float scaleY = 0.4f;
        float x = 5;
        float y_new = -15;
        if (visible) {
            batch.draw(texture, x, y_new, texture.getWidth() * scaleX, texture.getHeight() * scaleY);
        }
    }

    public boolean isUnlocked() {
        return unlocked;
    }

    public void dispose() {
        texture.dispose();
    }
}
