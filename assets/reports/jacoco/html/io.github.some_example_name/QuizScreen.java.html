package io.github.some_example_name;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Input;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.GL20;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.BitmapFont;
import com.badlogic.gdx.graphics.g2d.GlyphLayout;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.math.Rectangle;

public class QuizScreen implements Screen {

    private final MyGame game;
    private final GameScreen gameScreen;
    private final QuizLogic logic;

    private Texture background;
    private SpriteBatch batch;
    private BitmapFont font;
    private GlyphLayout layout;

    public Rectangle button1, button2, button3, button4;

    public QuizScreen(MyGame game, GameScreen gameScreen) {
        this.game = game;
        this.gameScreen = gameScreen;
        this.logic = new QuizLogic(3f);

        background = new Texture("quiz_screen_final.png");
        batch = new SpriteBatch();

        font = new BitmapFont();
        font.getData().setScale(8f);
        layout = new GlyphLayout();

        float w = Gdx.graphics.getWidth();
        float h = Gdx.graphics.getHeight();

        button1 = new Rectangle(w * 0.22917f, h * 0.39556f, w * 0.22222f, h * 0.225f);
        button2 = new Rectangle(w * 0.52986f, h * 0.39556f, w * 0.22222f, h * 0.225f);
        button3 = new Rectangle(w * 0.22917f, h * 0.09722f, w * 0.22222f, h * 0.225f);
        button4 = new Rectangle(w * 0.52986f, h * 0.08889f, w * 0.22222f, h * 0.225f);
    }

    @Override
    public void render(float delta) {
        Gdx.gl.glClearColor(0, 0, 0, 1);
        Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT);

        float mouseX = Gdx.input.getX();
        float mouseY = Gdx.graphics.getHeight() - Gdx.input.getY();

        if (!logic.isAnswered() && Gdx.input.isButtonJustPressed(Input.Buttons.LEFT)) {
            boolean correct =
                button2.contains(mouseX, mouseY);

            if (button1.contains(mouseX, mouseY)
                || button3.contains(mouseX, mouseY)
                || button4.contains(mouseX, mouseY)
                || correct) {

                QuizLogic.AnswerResult result = logic.answer(correct);

                if (!logic.isEventTriggered()) {
                    if (result == QuizLogic.AnswerResult.CORRECT) {
                        gameScreen.onQuizAnsweredCorrectly();
                    } else if (result == QuizLogic.AnswerResult.INCORRECT) {
                        gameScreen.onQuizAnsweredIncorrectly();
                    }
                    logic.markEventTriggered();
                }
            }
        }

        batch.begin();

        if (!logic.isAnswered()) {
            batch.draw(background, 0, 0,
                Gdx.graphics.getWidth(),
                Gdx.graphics.getHeight());
        } else {
            String text = logic.isCorrect() ? "Correct" : "Incorrect";
            layout.setText(font, text);
            font.draw(batch, layout,
                (Gdx.graphics.getWidth() - layout.width) / 2,
                (Gdx.graphics.getHeight() + layout.height) / 2);
        }

        batch.end();

        if (logic.update(delta)) {
            game.goToGame();
        }

        if (Gdx.input.isKeyJustPressed(Input.Keys.E)) {
            Gdx.app.exit();
        }
    }

    @Override public void show() {}
    @Override public void resize(int width, int height) {}
    @Override public void pause() {}
    @Override public void resume() {}
    @Override public void hide() {}

    @Override
    public void dispose() {
        batch.dispose();
        font.dispose();
        background.dispose();
    }
}
