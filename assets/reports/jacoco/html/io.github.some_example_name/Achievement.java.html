package io.github.some_example_name;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Input;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;

/**
 * Handles achievement rendering and input.
 */
public class Achievement {

    private final Texture texture;
    private final AchievementLogic logic;

    private static Achievement activeAchievement;

    public Achievement(String image) {
        texture = new Texture(image);
        logic = new AchievementLogic(
            -texture.getHeight(), // start off-screen
            20f,                  // target Y
            2000f                 // animation speed
        );
    }

    /** Unlocks and shows the achievement */
    public void unlock() {
        if (activeAchievement != null && activeAchievement != this) {
            activeAchievement.invisible();
        }
        activeAchievement = this;
        logic.unlock();
    }

    /** Shows achievement without unlocking */
    public void show() {
        if (activeAchievement != null && activeAchievement != this) {
            activeAchievement.invisible();
        }
        activeAchievement = this;
        logic.unlock();
    }

    /** Hides achievement */
    public void invisible() {
        logic.hide();
    }

    /** Updates animation and input */
    public void update(float delta) {
        logic.update(delta);

        if (logic.isVisible() && Gdx.input.isKeyJustPressed(Input.Keys.SPACE)) {
            logic.hide();
        }
    }

    /** Draws achievement if visible */
    public void draw(SpriteBatch batch, float screenWidth) {
        if (!logic.isVisible()) return;

        float scaleX = 0.2f;
        float scaleY = 0.4f;

        batch.draw(
            texture,
            5,
            -15,
            texture.getWidth() * scaleX,
            texture.getHeight() * scaleY
        );
    }

    public boolean isUnlocked() {
        return logic.isUnlocked();
    }

    public void dispose() {
        texture.dispose();
    }
}
